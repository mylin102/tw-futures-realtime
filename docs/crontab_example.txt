# ============================================
# 台指期交易系統
# ============================================

# 🌙 夜盤啟動 (週一至週五 14:55)

0 15 * * 1-5 /Users/mylin/Documents/mylin102/tw-futures-realtime/autostart.sh

# ☀️ 日盤啟動 (週一至週五 08:40)
# 提前 5 分鐘啟動，確保系統就緒
# 同時啟動交易系統和儀表板
40 8 * * 1-5 /Users/mylin/Documents/mylin102/tw-futures-realtime/autostart.sh

# ============================================
# 系統維護
# ============================================

# 每日凌晨 4:00 清理舊日誌 (保留 7 天)
0 4 * * * find /Users/mylin/Documents/mylin102/tw-futures-realtime/logs -name "*.log" -mtime +7 -delete

# 每週一 09:00 發送週報
0 9 * * 1 /Users/mylin/Documents/mylin102/tw-futures-realtime/scripts/weekly_report.py

# ============================================
# 監控
# ============================================

# 每 5 分鐘檢查交易系統狀態
*/5 * * * * ps aux | grep -v grep | grep "daily_simulation\|night_trading" > /dev/null || echo "[$(date)] Trading system not running" >> /Users/mylin/Documents/mylin102/tw-futures-realtime/logs/monitor.log

# 每小時檢查儀表板狀態
0 * * * * ps aux | grep -v grep | grep "streamlit.*dashboard" > /dev/null || echo "[$(date)] Dashboard not running" >> /Users/mylin/Documents/mylin102/tw-futures-realtime/logs/monitor.log
